44 template <typename MatsT, typename IntsT>
45 template <typename MatsU>
46 std::shared_ptr<ERIContractions<MatsU,IntsT>>
47 ERIContractions<MatsT,IntsT>::convert(const std::shared_ptr<ERIContractions<MatsT,IntsT>>& ch) {
48
49 if (not ch) return nullptr;
50
51 const std::type_info &tID(typeid(*ch));
52
53 if (tID == typeid(InCore4indexERIContraction<MatsT,IntsT>)) {
54 return std::make_shared<InCore4indexERIContraction<MatsU,IntsT>>(
55 *std::dynamic_pointer_cast<InCore4indexERIContraction<MatsT,IntsT>>(ch));
56
57 } else if (tID == typeid(GTODirectERIContraction<MatsT,IntsT>)) {
58 return std::make_shared<GTODirectERIContraction<MatsU,IntsT>>(
59 *std::dynamic_pointer_cast<GTODirectERIContraction<MatsT,IntsT>>(ch));
60
61 } else if (tID == typeid(GIAODirectERIContraction)) {
62 return std::dynamic_pointer_cast<ERIContractions<MatsU,IntsT>>(ch);
63
64 } else if (tID == typeid(InCoreRIERIContraction<MatsT,IntsT>)) {
65 return std::make_shared<InCoreRIERIContraction<MatsU,IntsT>>(
66 *std::dynamic_pointer_cast<InCoreRIERIContraction<MatsT,IntsT>>(ch));
67
68 } else {
69 std::stringstream errMsg;
70 errMsg << "ERIContractions implementation \"" << tID.name() << "\" not registered in convert." << std::endl;
71 CErr(errMsg.str(),std::cout);
72 }
73
74 return nullptr;
75
76 }
