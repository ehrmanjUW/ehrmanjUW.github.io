158 template <size_t N>
159 inline std::array<double, N> getNAmp(EMFieldTyp TYPE) {
160
161 std::array<double, N> amp = {0., 0., 0.};
162
163
164 for(auto &field : fields) {
165
166 if( field->emFieldTyp == TYPE and field->size == N ) {
167
168 auto tmp_amp = field->getAmp();
169
170 for( auto k = 0; k < N; k++) amp[k] += tmp_amp[k];
171
172 }
173
174 }
175
176 return amp;
177
178 }
