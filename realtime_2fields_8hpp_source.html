196 EMPerturbation getPert(double t) {
197
198 EMPerturbation pert;
199
200 for(auto &field : fields) {
201 auto amp = field->getAmp(t);
202
203 if(amp.size() == 3) {
204 TDDipoleField &fld = dynamic_cast<TDDipoleField&>(*field);
205 std::array<double,3> dple = {amp[0], amp[1], amp[2]};
206
207 pert.addField(fld.emFieldTyp,fld.fieldGauge,dple);
208 }
209
210 }
211
212 return pert;
213
214 }

218 template <size_t N>
219 inline std::array<double, N> getNAmp(EMFieldTyp TYPE, double t) {
220
221 std::array<double, N> amp = {0., 0., 0.};
222
223
224 for(auto &field : fields) {
225
226 if( field->emFieldTyp == TYPE and field->size == N ) {
227
228 auto tmp_amp = field->getAmp(t);
229 amp[0] += tmp_amp[0];
230 amp[1] += tmp_amp[1];
231 amp[2] += tmp_amp[2];
232
233 }
234
235 }
236
237 return amp;
238 }

251 bool isFieldDiscontinuous(double curT, double deltaT) {
252
253 bool discont = false;
254 for ( auto &field : fields ) {
255 discont |= (curT >= field->envelope->tOn and curT - deltaT < field->envelope->tOn);
256 discont |= (curT >= field->envelope->tOff and curT - deltaT < field->envelope->tOff);
257 }
