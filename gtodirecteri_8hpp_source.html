30 template <typename IntsT>
31 class DirectERI : public TwoEInts<IntsT> {
32
33 template <typename IntsU>
34 friend class DirectERI;
35
36 protected:
37 BasisSet &basisSet_;
38 double threshSchwartz_ = 1e-12;
39 double* schwartz_ = nullptr;
40
41 public:
42
43 // Constructor
44 DirectERI() = delete;
45 DirectERI(CQMemManager &mem, BasisSet &basis, double threshSchwartz):
46 TwoEInts<IntsT>(mem, basis.nBasis), basisSet_(basis),
47 threshSchwartz_(threshSchwartz) {}
48 DirectERI( const DirectERI &other ):
49 DirectERI(other.memManager(), other.nBasis(), other.threshSchwartz_) {
50 std::copy_n(other.schwartz_, basisSet_.nShell*basisSet_.nShell, schwartz_);
51 }
52 template <typename IntsU>
53 DirectERI( const DirectERI<IntsU> &other, int = 0 ):
54 DirectERI(other.memManager_, other.basisSet_, other.threshSchwartz_) {
55 if (other.schwartz_) {
56 size_t NS = basisSet().nShell;
57 schwartz_ = this->memManager().template malloc<double>(NS*NS);
58 std::copy_n(other.schwartz_, NS*NS, schwartz_);
59 }
60 }
61 DirectERI( DirectERI &&other ): TwoEInts<IntsT>(std::move(other)),
62 basisSet_(other.basis), threshSchwartz_(other.threshSchwartz_),
63 schwartz_(other.schwartz_) { other.schwartz_ = nullptr; }
64
65 BasisSet& basisSet() { return basisSet_; }
66 double threshSchwartz() const { return threshSchwartz_; }
67 double*& schwartz() { return schwartz_; }
