154 oeiStr = "Relativistic one-electron integral";

157 out << oeiStr;

166 template <typename TransT>
167 OneERelInts<typename std::conditional<
168 (std::is_same<IntsT, dcomplex>::value or
169 std::is_same<TransT, dcomplex>::value),
170 dcomplex, double>::type> transform(
171 char TRANS, const TransT* T, int NT, int LDT) const {
172 OneERelInts<typename std::conditional<
173 (std::is_same<IntsT, dcomplex>::value or
174 std::is_same<TransT, dcomplex>::value),
175 dcomplex, double>::type> transInts(
176 this->ElectronIntegrals::memManager(), NT, hasSpinOrbit());
177 transInts[REL_INTS_COMPS::O] =
178 (*this)[REL_INTS_COMPS::O].transform(TRANS, T, NT, LDT);
179 transInts.scalar() = scalar().transform(TRANS, T, NT, LDT);
180 if (hasSpinOrbit()) {
181 transInts.SOX() = SOX().transform(TRANS, T, NT, LDT);
182 transInts.SOY() = SOY().transform(TRANS, T, NT, LDT);
183 transInts.SOZ() = SOZ().transform(TRANS, T, NT, LDT);
184 }
185 return transInts;
186 }
