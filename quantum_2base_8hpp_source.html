141 void computeEnergy(EMPerturbation &pert){
142
143 ROOT_ONLY(comm);
144
145 computeEnergy();
146
147
148 // Field contributions to the energy
149 double field_delta(0.);
150
151 // Purely magnetic field contributions here
152 auto magDipoleField = pert.getDipoleAmp(Magnetic);
153
154
155 // Purely electric field contributions here
156 auto elecDipoleField = pert.getDipoleAmp(Electric);
157 field_delta +=
158 elecDipoleField[0] * elecDipole[0] +
159 elecDipoleField[1] * elecDipole[1] +
160 elecDipoleField[2] * elecDipole[2];
161
162
163 totalEnergy += field_delta; // Increment total energy
164
165 };
