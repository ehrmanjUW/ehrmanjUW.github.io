90 template <typename... Args>
91 KohnSham(std::string funcName,
92 std::vector<std::shared_ptr<DFTFunctional>> funclist,
93 MPI_Comm c, IntegrationParam ip,
94 CQMemManager &mem, Molecule &mol, BasisSet &basis,
95 Integrals<IntsT> &aoi, Args... args) :
96 SingleSlater<MatsT,IntsT>(c,mem,mol,basis,aoi,args...),
97 WaveFunctionBase(c,mem,args...),
98 QuantumBase(c,mem,args...), isGGA_(false),
99 functionals(std::move(funclist)),intParam(ip){
100
101 // Append HF tags to reference names
102 if(this->nC == 1) {
103 if(this->iCS) {
104 this->refLongName_ += "Restricted " + funcName;
105 this->refShortName_ += "R" + funcName;
106 } else {
107 this->refLongName_ += "Unrestricted " + funcName;
108 this->refShortName_ += "U" + funcName;
109 }
110 } else {
111 this->refLongName_ += "Generalized " + funcName;
112 this->refShortName_ += "G" + funcName;
113 }
114
115 size_t NB = this->basisSet().nBasis;
116 if(this->nC > 1)
117 VXC = std::make_shared<PauliSpinorSquareMatrices<double>>(this->memManager, NB, true);
118 else if (not this->iCS)
119 VXC = std::make_shared<PauliSpinorSquareMatrices<double>>(this->memManager, NB, false);
120 else
121 VXC = std::make_shared<PauliSpinorSquareMatrices<double>>(this->memManager, NB, false, false);
122
123
124 }; // KohnSham constructor

168 virtual void formFock(EMPerturbation &pert, bool increment = false, double HFX = 0.) {
169
170 SingleSlater<MatsT,IntsT>::formFock(pert,increment,functionals.back()->xHFX);
171
172 auto VXCStart = tick();
173 formVXC();
174 VXCDur = tock(VXCStart);
175
176 ROOT_ONLY(this->comm);
177
178 // Add VXC in Fock matrix
179 *this->fockMatrix += *VXC;
180
181 }; // formFock
182
188 virtual void computeEnergy() {
189
190 SingleSlater<MatsT,IntsT>::computeEnergy();
191 // Add EXC in the total energy
192 this->totalEnergy += XCEnergy;
193
194 }; // computeEnergy
