64 template <typename MatsT, typename IntsT>
65 void FockBuilder<MatsT,IntsT>::formGD(SingleSlater<MatsT,IntsT> &ss,
66 EMPerturbation &pert, bool increment, double xHFX) {
67
68 typedef MatsT* oper_t;
69 typedef std::vector<oper_t> oper_t_coll;
70
71 // Decide list of onePDMs to use
72 PauliSpinorSquareMatrices<MatsT> &contract1PDM
73 = increment ? *ss.deltaOnePDM : *ss.onePDM;
74
75 size_t NB = ss.basisSet().nBasis;
76
77 // Zero out J and K[i]
78 if(not increment) {
79 ss.coulombMatrix->clear();
80 ss.exchangeMatrix->clear();
81 }
82
83 std::vector<TwoBodyContraction<MatsT>> contract =
84 { {contract1PDM.S().pointer(), ss.coulombMatrix->pointer(), true, COULOMB} };
85
86 // Determine how many (if any) exchange terms to calculate
87 if( std::abs(xHFX) > 1e-12 and not increment and
88 (ss.scfControls.guess != SAD or ss.scfConv.nSCFIter != 0) and
89 std::dynamic_pointer_cast<InCoreRIERIContraction<MatsT, IntsT>>(ss.ERI)) {
90 ROOT_ONLY(ss.comm);
91 auto rieri = std::dynamic_pointer_cast<InCoreRIERIContraction<MatsT, IntsT>>(ss.ERI);
92
93 if(ss.nC == 1) {
94 SquareMatrix<MatsT> AAblock(ss.exchangeMatrix->memManager(), NB);
95 rieri->KCoefContract(ss.comm, ss.nOA, ss.mo[0].pointer(), AAblock.pointer());
96 if(ss.iCS) {
97 *ss.exchangeMatrix = PauliSpinorSquareMatrices<MatsT>::spinBlockScatterBuild(AAblock);
98 } else {
99 SquareMatrix<MatsT> BBblock(ss.exchangeMatrix->memManager(), NB);
100 rieri->KCoefContract(ss.comm, ss.nOB, ss.mo[1].pointer(), BBblock.pointer());
101 *ss.exchangeMatrix = PauliSpinorSquareMatrices<MatsT>::spinBlockScatterBuild(AAblock, BBblock);
102 }
103 } else {
104 SquareMatrix<MatsT> spinBlockForm(ss.exchangeMatrix->memManager(), NB*2);
105 InCoreRIERI<MatsT> rierispinor(dynamic_cast<InCoreRIERI<IntsT>&>(rieri->ints()).
106 template spatialToSpinBlock<MatsT>());
107 InCoreRIERIContraction<MatsT,MatsT>(rierispinor)
108 .KCoefContract(ss.comm, ss.nO, ss.mo[0].pointer(), spinBlockForm.pointer());
109 *ss.exchangeMatrix = spinBlockForm.template spinScatter<MatsT>();
110 }
111
112 } else if( std::abs(xHFX) > 1e-12 ) {
113 contract.push_back(
114 {contract1PDM.S().pointer(), ss.exchangeMatrix->pointer(), true, EXCHANGE}
115 );
116 if (ss.exchangeMatrix->hasZ())
117 contract.push_back(
118 {contract1PDM.Z().pointer(), ss.exchangeMatrix->Z().pointer(), true, EXCHANGE}
119 );
120 if (ss.exchangeMatrix->hasXY()) {
121 contract.push_back(
122 {contract1PDM.Y().pointer(), ss.exchangeMatrix->Y().pointer(), true, EXCHANGE}
123 );
124 contract.push_back(
125 {contract1PDM.X().pointer(), ss.exchangeMatrix->X().pointer(), true, EXCHANGE}
126 );
127 }
128 }
129
130 ss.ERI->twoBodyContract(ss.comm, contract, pert);
131
132 ROOT_ONLY(ss.comm); // Return if not root (J/K only valid on root process)
133
134
135 // Form GD: G[D] = 2.0*J[D] - K[D]
136 if( std::abs(xHFX) > 1e-12 ) {
137 *ss.twoeH = -xHFX * *ss.exchangeMatrix;
138 } else {
139 ss.twoeH->clear();
140 }
141 // G[D] += 2*J[D]
142 *ss.twoeH += 2.0 * *ss.coulombMatrix;
143
144#if 0
145 //printJ(std::cout);
146 printK(std::cout);
147 //printGD(std::cout);
148#endif
149
150 } // FockBuilder::formGD

161 template <typename MatsT, typename IntsT>
162 void FockBuilder<MatsT,IntsT>::formFock(SingleSlater<MatsT,IntsT> &ss,
163 EMPerturbation &pert, bool increment, double xHFX) {
164
165 size_t NB = ss.basisSet().nBasis;
166 size_t NB2 = NB*NB;
167
168 auto GDStart = tick(); // Start time for G[D]
169
170 // Form G[D]
171 formGD(ss,pert,increment,xHFX);
172
173 ss.GDDur = tock(GDStart); // G[D] Duraction
174
175 ROOT_ONLY(ss.comm);
176
177 // Form Fock
178 *ss.fockMatrix = *ss.coreH + *ss.twoeH;
179
180 // Add in the electric field contributions
181 // FIXME: the magnetic field contribution should go here as well to allow for RT
182 // manipulation
183
184 if( pert_has_type(pert,Electric) ) {
185
186 auto dipAmp = pert.getDipoleAmp(Electric);
187
188 for(auto i = 0; i < 3; i++)
189 ss.fockMatrix->S() -=
190 2. * dipAmp[i] * (*ss.aoints.lenElectric)[i].matrix();
191
192 }
193
194#if 0
195 ss.printFock(std::cout);
196#endif
197 }

208 FockBuilder<MatsT,IntsT>::convert(const std::shared_ptr<FockBuilder<MatsT,IntsT>>& fb) {
209
210 if (not fb) return nullptr;
211
212 const std::type_info &tID(typeid(*fb));
213
214 if (tID == typeid(ROFock<MatsT,IntsT>)) {
215 return std::make_shared<ROFock<MatsU,IntsT>>(
216 *std::dynamic_pointer_cast<ROFock<MatsT,IntsT>>(fb));
217
218 } else {
219 return std::make_shared<FockBuilder<MatsU,IntsT>>(
220 *std::dynamic_pointer_cast<FockBuilder<MatsT,IntsT>>(fb));
221 }
