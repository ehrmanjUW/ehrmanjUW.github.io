38 void SingleSlaterBase::SCF(EMPerturbation &pert) {
39
40 SCFInit();
41
42 // Initialize type independent parameters
43 bool isConverged = false;
44 scfControls.dampParam = scfControls.dampStartParam;
45
46//XSLIC
47// scfControls.doIncFock = scfControls.doIncFock and (this->aoints.cAlg == DIRECT);
48 scfControls.doIncFock = false;
49
50 if( scfControls.scfAlg == _NEWTON_RAPHSON_SCF )
51 scfControls.doExtrap = false;
52
53 if( scfControls.scfAlg == _SKIP_SCF )
54 isConverged = true;
55
56 // Compute initial properties
57 this->computeProperties(pert);
58
59
60 if( printLevel > 0 and MPIRank(comm) == 0 ) {
61 printSCFHeader(std::cout,pert);
62 printSCFProg(std::cout,false);
63 }
64
65 for( scfConv.nSCFIter = 0; scfConv.nSCFIter < scfControls.maxSCFIter;
66 scfConv.nSCFIter++) {
67
68 // Save current state of the wave function (method specific)
69 saveCurrentState();
70
71 // Exit loop on convergence
72 if(isConverged) break;
73
74 // Get new orbtials and densities from current state:
75 // C/D(k) -> C/D(k + 1)
76 getNewOrbitals(pert);
77
78 // Evaluate convergence
79 isConverged = evalConver(pert);
80
81 // Print out iteration information
82 if( printLevel > 0 and (MPIRank(comm) == 0)) printSCFProg(std::cout);
83
84 }; // Iteration loop
85
86 // Save current state of the wave function (method specific)
87 saveCurrentState();
88
89 SCFFin();
90
91 // Compute initial properties
92 this->computeProperties(pert);
93
94 //printSCFFooter(isConverged);
95 if(not isConverged)
96 CErr(std::string("SCF Failed to converged within ") +
97 std::to_string(scfControls.maxSCFIter) +
98 std::string(" iterations"));
99 else if( printLevel > 0 ) {
100 std::cout << std::endl << "SCF Completed: E("
101 << refShortName_ << ") = " << std::fixed
102 << std::setprecision(10) << this->totalEnergy
103 << " Eh after " << scfConv.nSCFIter
104 << " SCF Iterations" << std::endl;
105 }
106
107 if( printLevel > 0 ) std::cout << BannerEnd << std::endl;
108
109 if( printLevel > 0 ) {
110 this->printMOInfo(std::cout);
111 this->printMultipoles(std::cout);
112 this->printSpin(std::cout);
113 this->printMiscProperties(std::cout);
114 }
115
116 }; // SingleSlaterBase::SCF()

119 void SingleSlaterBase::printSCFHeader(std::ostream &out,
120 EMPerturbation &pert) {
121
122 out << BannerTop << std::endl;
123 out << "Self Consistent Field (SCF) Settings:" << std::endl << std::endl;
124
125 out << std::setw(38) << std::left << " Reference:" << refLongName_
126 << std::endl;
127
128 out << std::setprecision(6) << std::scientific;
129 out << std::setw(38) << std::left << " Density Convergence Tolerence:"
130 << scfControls.denConvTol << std::endl;
131
132 out << std::setw(38) << std::left << " Energy Convergence Tolerence:"
133 << scfControls.eneConvTol << std::endl;
134
135 out << std::setw(38) << std::left << " Maximum Number of SCF Cycles:"
136 << scfControls.maxSCFIter << std::endl;
137
138
139 out << std::setw(38) << std::left << " SCF Algorithm:";
140 if( scfControls.scfAlg == _CONVENTIONAL_SCF )
141 out << "Conventional SCF";
142 else if( scfControls.scfAlg == _NEWTON_RAPHSON_SCF )
143 out << "Newton-Raphson (2nd Order)";
144 out << "\n";
145
146
147 if (scfControls.doExtrap) {
148 if (scfControls.doDamp) {
149 out << std::setw(38) << std::left << " Static Damping Factor:"
150 << scfControls.dampParam << std::endl;
151 out << std::setw(38) << std::left << " Damping Error:"
152 << scfControls.dampError << std::endl;
153 }
154
155 if (scfControls.diisAlg != NONE) {
156 out << std::setw(38) << std::left << " DIIS Extrapolation Algorithm:";
157 if (scfControls.diisAlg == CDIIS) out << "CDIIS";
158 out << std::endl;
159
160 out << std::left << " * DIIS will track up to "
161 << scfControls.nKeep << " previous iterations" << std::endl;
162 }
163
164 }
165
166
167 if( scfControls.doIncFock ) {
168 out << "\n * Will Perform Incremental Fock Build -- Restarting Every "
169 << scfControls.nIncFock << " SCF Steps\n";
170
171 }
172
173 // Field print
174 if( pert.fields.size() != 0 ) {
175
176 out << "\n\n * SCF will be performed in the presence of an EM "
177 << "perturbation:\n\n";
178
179 for(auto &field : pert.fields) {
180
181 auto amp = field->getAmp();
182
183 out << " * ";
184 if( field->emFieldTyp == Electric ) out << "Electric";
185 else out << "Magnetic";
186
187 out << " ";
188
189 if( field->size == 3 ) out << "Dipole";
190 else if ( field->size == 6 ) out << "Quadrupole";
191 else if ( field->size == 10 ) out << "Octupole";
192
193 out << " Field: ";
194 out << "{ ";
195 for(auto i = 0; i < amp.size(); i++) {
196 out << amp[i]; if(i != amp.size() - 1) out << ", ";
197 }
198 out << " }\n";
199
200 }
201
202
203 }
204
205
206
207 out << std::endl << BannerMid << std::endl << std::endl;
208 out << std::setw(16) << "SCF Iteration";
209 out << std::setw(18) << "Energy (Eh)";
210 out << std::setw(18) << "\u0394E (Eh)";
211 out << std::setw(18) << " |\u0394P(S)|";
212 if(not iCS or nC > 1)
213 out << std::setw(18) << " |\u0394P(M)|";
214
215 out << std::endl;
216 out << std::setw(16) << "-------------";
217 out << std::setw(18) << "-----------";
218 out << std::setw(18) << "-------";
219 out << std::setw(18) << "-------";
220 if(not iCS or nC > 1)
221 out << std::setw(18) << "-------";
222 out << std::endl;
223
224 }; // SingleSlater<T>::printSCFHeader

231 void SingleSlaterBase::printSCFProg(std::ostream &out,
232 bool printDiff) {
233
234 // SCF Iteration
235 out << " SCFIt: " <<std::setw(6) << std::left;
236
237 if( printDiff ) out << scfConv.nSCFIter + 1;
238 else out << 0;
239
240 // Current Total Energy
241 out << std::setw(18) << std::fixed << std::setprecision(10)
242 << std::left << this->totalEnergy;
243
244 if( printDiff ) {
245 out << std::scientific << std::setprecision(7);
246 // Current Change in Energy
247 out << std::setw(14) << std::right << scfConv.deltaEnergy;
248 out << " ";
249 out << std::setw(13) << std::right << scfConv.RMSDenScalar;
250 if(not iCS or nC > 1) {
251 out << " ";
252 out << std::setw(13) << std::right << scfConv.RMSDenMag;
253 }
254 }
255
256 out << std::endl;
257 }; // SingleSlater<T>::printSCFProg
