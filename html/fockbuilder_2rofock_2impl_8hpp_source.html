39 template <typename MatsT, typename IntsT>
40 void ROFock<MatsT,IntsT>::rohfFock(SingleSlater<MatsT,IntsT> &ss) {
41
42 size_t NB = ss.basisSet().nBasis;
43 size_t NB2 = NB*NB;
44
45 ROOT_ONLY(ss.comm);
46
47 //construct focka and fockb
48 ss.mo = ss.fockMatrix->template spinGatherToBlocks<MatsT>(false);
49
50 //construct projectors for closed, open, virtual
51 //pc = dmb * S
52 //po = (dma - dmb) * S
53 //pv = I - dma * S
54 MatsT* pc = ss.memManager.template malloc<MatsT>(NB*NB);
55 MatsT* po = ss.memManager.template malloc<MatsT>(NB*NB);
56 MatsT* pv = ss.memManager.template malloc<MatsT>(NB*NB);
57 SquareMatrix<MatsT> tmp(ss.memManager,NB);
58 MatsT* tmp2 = ss.memManager.template malloc<MatsT>(NB*NB);
59
60 //overlap matrix
61 tmp = 0.5 * (ss.onePDM->S() - ss.onePDM->Z());
62 Gemm('N','N',NB,NB,NB,MatsT(1.),tmp.pointer(),NB,
63 SquareMatrix<MatsT>(ss.aoints.overlap->matrix()).pointer(),NB,
64 MatsT(0.),pc,NB);
65 Gemm('N','N',NB,NB,NB,MatsT(1.),ss.onePDM->Z().pointer(),NB,
66 SquareMatrix<MatsT>(ss.aoints.overlap->matrix()).pointer(),NB,
67 MatsT(0.),po,NB);
68 tmp = 0.5 * (ss.onePDM->S() + ss.onePDM->Z());
69 Gemm('N','N',NB,NB,NB,MatsT(-1.),tmp.pointer(),NB,
70 SquareMatrix<MatsT>(ss.aoints.overlap->matrix()).pointer(),NB,
71 MatsT(0.),pv,NB);
72 for(auto j = 0; j < NB; j++) pv[j*NB+j] = MatsT(1.) + pv[j*NB+j];
73 /*
74 * construct Roothaan's effective fock
75 ======== ======== ====== =========
76 space closed open virtual
77 ======== ======== ====== =========
78 closed Fc Fb Fc
79 open Fb Fc Fa
80 virtual Fc Fa Fc
81 ======== ======== ====== =========
82 where Fc = (Fa + Fb) / 2
83 */
84 Gemm('N','N',NB,NB,NB,MatsT(0.5),ss.fockMatrix->S().pointer(),NB,
85 pc,NB,MatsT(0.),tmp2,NB);
86 Gemm('C','N',NB,NB,NB,MatsT(0.5),pc,NB,tmp2,NB,MatsT(0.),tmp.pointer(),NB);
87 Gemm('C','N',NB,NB,NB,MatsT(1.),pv,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
88 Gemm('N','N',NB,NB,NB,MatsT(0.5),ss.fockMatrix->S().pointer(),NB,
89 po,NB,MatsT(0.),tmp2,NB);
90 Gemm('C','N',NB,NB,NB,MatsT(0.5),po,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
91 Gemm('N','N',NB,NB,NB,MatsT(0.5),ss.fockMatrix->S().pointer(),NB,
92 pv,NB,MatsT(0.),tmp2,NB);
93 Gemm('C','N',NB,NB,NB,MatsT(0.5),pv,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
94 Gemm('N','N',NB,NB,NB,MatsT(1.),ss.mo[1].pointer(),NB,pc,NB,MatsT(0.),tmp2,NB);
95 Gemm('C','N',NB,NB,NB,MatsT(1.),po,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
96 Gemm('N','N',NB,NB,NB,MatsT(1.),ss.mo[0].pointer(),NB,pv,NB,MatsT(0.),tmp2,NB);
97 Gemm('C','N',NB,NB,NB,MatsT(1.),po,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
98 MatAdd('C','N',NB,NB,MatsT(1.),tmp.pointer(),NB,MatsT(1.),
99 tmp.pointer(),NB,ss.fockMatrix->S().pointer(),NB);
100
101 ss.memManager.free(pc,po,pv,tmp2);
102 }; // ROFock<MatsT, IntsT>::rohfFock
