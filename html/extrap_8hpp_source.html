37 template <typename MatsT, typename IntsT>
38 void SingleSlater<MatsT,IntsT>::modifyFock() {
39
40 ROOT_ONLY(comm);
41
42 // Static Damping
43 scfControls.doDamp &= (std::dynamic_pointer_cast<ROFock<MatsT,IntsT>>(fockBuilder) == nullptr);
44 if (scfControls.doDamp) fockDamping();
45
46 // DIIS extrapolation
47 if (scfControls.diisAlg == NONE) return;
48
49 if (scfControls.diisAlg == CDIIS) {
50 size_t nExtrap = std::min(scfConv.nSCFIter+1,scfControls.nKeep);
51 scfDIIS(nExtrap);
52 } else CErr("Only CDIIS is implemented so far",std::cout);
53
54 }; // SingleSlater<T>::modifyFock

64 template <typename MatsT, typename IntsT>
65 void SingleSlater<MatsT,IntsT>::fockDamping() {
66
67 // Don't damp the first iteration. We don't want
68 // to use the guess Fock and it's not saved anyway.
69 if(scfConv.nSCFIter == 0) return;
70
71 size_t NB = basisSet().nBasis;
72 double dp = scfControls.dampParam;
73
74 // Damp the current orthonormal Fock matrix
75 if( not savFile.exists() )
76 *fockMatrixOrtho = (1-dp) * *fockMatrixOrtho + dp * *prevFock;
77 else {
78
79 PauliSpinorSquareMatrices<MatsT> FSCR(this->memManager, NB,
80 fockMatrixOrtho->hasXY(), fockMatrixOrtho->hasZ());
81
82 savFile.readData("/SCF/FOCK_ORTHO", FSCR);
83
84 *fockMatrixOrtho = (1-dp) * *fockMatrixOrtho + dp * FSCR;
85
86 }
87
88 }; // SingleSlater<T>::fockDamping

98 template <typename MatsT, typename IntsT>
99 void SingleSlater<MatsT,IntsT>::scfDIIS(size_t nExtrap) {
100
101 // Save the current AO Fock and density matrices
102 size_t NB = basisSet().nBasis;
103 size_t iDIIS = scfConv.nSCFIter % scfControls.nKeep;
104
105 diisFock[iDIIS] = *this->fockMatrix;
106 diisOnePDM[iDIIS] = *this->onePDM;
107
108 // Evaluate orthonormal [F,D] and store in diisError
109 FDCommutator(diisError[iDIIS]);
110
111 scfConv.nrmFDC = 0.;
112 for(auto E : diisError[iDIIS].SZYXPointers())
113 scfConv.nrmFDC = std::max(scfConv.nrmFDC,TwoNorm<double>(NB*NB,E,1));
114
115 // Just save the Fock, density, and commutator for the first iteration
116 if (scfConv.nSCFIter == 0) return;
117
118 // Build the B matrix and return the coefficients for the extrapolation
119 DIIS<MatsT> extrap(nExtrap,diisError);
120
121
122 if(extrap.extrapolate()) {
123 // Extrapolate Fock and density matrices using DIIS coefficients
124 fockMatrix->clear();
125 this->onePDM->clear();
126 for(auto j = 0; j < nExtrap; j++) {
127 *fockMatrix += extrap.coeffs[j] * diisFock[j];
128 *this->onePDM += extrap.coeffs[j] * diisOnePDM[j];
129 }
130 } else {
131 std::cout << "\n *** WARNING: DIIS Inversion Failed -- "
132 << " Defaulting to Fixed-Point step ***\n" << std::endl;
133 }
134
135 // Transform AO fock into the orthonormal basis
136 ao2orthoFock();
137
138 }; // SingleSlater<T>::scfDIIS

146 template <typename MatsT, typename IntsT>
147 void SingleSlater<MatsT,IntsT>::allocExtrapStorage() {
148
149 ROOT_ONLY(comm);
150
151 diisFock.clear();
152 diisFock.reserve(scfControls.nKeep);
153 diisOnePDM.clear();
154 diisOnePDM.reserve(scfControls.nKeep);
155 diisError.clear();
156 diisError.reserve(scfControls.nKeep);
157
158 // Allocate memory to store previous orthonormal Focks and densities for DIIS
159 if (scfControls.diisAlg != NONE) {
160 for(auto i = 0; i < scfControls.nKeep; i++) {
161 diisFock.emplace_back(memManager, fockMatrix->dimension(),
162 fockMatrix->hasXY(), fockMatrix->hasZ());
163 diisFock.back().clear();
164 diisOnePDM.emplace_back(memManager, fockMatrix->dimension(),
165 fockMatrix->hasXY(), fockMatrix->hasZ());
166 diisOnePDM.back().clear();
167 diisError.emplace_back(memManager, fockMatrix->dimension(),
168 fockMatrix->hasXY(), fockMatrix->hasZ());
169 diisError.back().clear();
170 }
171 }
172
173 // Allocate memory to store previous orthonormal Fock for damping
174 if( scfControls.doDamp and not savFile.exists() ) {
175 SPIN_OPERATOR_ALLOC(basisSet().nBasis,prevFock);
176 }
177
178 }; // SingleSlater<T>::allocExtrapStorage

214 template <typename MatsT, typename IntsT>
215 void SingleSlater<MatsT,IntsT>::FDCommutator(PauliSpinorSquareMatrices<MatsT> &FDC) {
216
217 size_t NB = basisSet().nBasis;
218 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT,IntsT>>(fockBuilder) != nullptr);
219
220 if(this->nC == 1) {
221 SquareMatrix<MatsT> SCR(memManager, NB);
222
223 // FD(S) = F(S)D(S)
224 Gemm('N', 'N', NB, NB, NB, MatsT(1.), fockMatrixOrtho->S().pointer(), NB,
225 onePDMOrtho->S().pointer(), NB, MatsT(0.), FDC.S().pointer(), NB);
226
227 // FD(S) += F(z)D(z)
228 if(nC == 2 or !iCS and !iRO) {
229 Gemm('N', 'N', NB, NB, NB, MatsT(1.), fockMatrixOrtho->Z().pointer(), NB,
230 onePDMOrtho->Z().pointer(), NB, MatsT(0.), SCR.pointer(), NB);
231 FDC.S() += SCR;
232 }
233
234 // Form {FD - DF}(S)
235 SCR = FDC.S();
236 MatAdd('N','C', NB, NB, MatsT(1.), FDC.S().pointer(), NB,
237 MatsT(-1.), SCR.pointer(), NB, FDC.S().pointer(), NB);
238
239
240 if(nC == 2 or !iCS and !iRO) {
241 // FD(z) = F(S)D(z) + F(z)D(S)
242 Gemm('N', 'N', NB, NB, NB, MatsT(1.), fockMatrixOrtho->S().pointer(), NB,
243 onePDMOrtho->Z().pointer(), NB, MatsT(0.), FDC.Z().pointer(), NB);
244 Gemm('N', 'N', NB, NB, NB, MatsT(1.), fockMatrixOrtho->Z().pointer(), NB,
245 onePDMOrtho->S().pointer(), NB, MatsT(0.), SCR.pointer(), NB);
246 FDC.Z() += SCR;
247
248 // Form {FD - DF}(z)
249 SCR = FDC.Z();
250 MatAdd('N','C', NB, NB, MatsT(1.), FDC.Z().pointer(), NB,
251 MatsT(-1.), SCR.pointer(), NB, FDC.Z().pointer(), NB);
252 }
253 } else {
254
255 // Gather the orthonormal Fock and densities
256 SquareMatrix<MatsT> FO(fockMatrixOrtho->template spinGather<MatsT>());
257 SquareMatrix<MatsT> DO(onePDMOrtho->template spinGather<MatsT>());
258 SquareMatrix<MatsT> SCR(memManager, 2*NB);
259
260 // Compute FD product
261 Gemm('N','N',2*NB,2*NB,2*NB,MatsT(1.),FO.pointer(),2*NB,
262 DO.pointer(),2*NB,MatsT(0.),SCR.pointer(),2*NB);
263
264 // Compute FD - DF (Store in FO scratch)
265 MatAdd('N','C',2*NB,2*NB,MatsT(1.),SCR.pointer(),2*NB,
266 MatsT(-1.),SCR.pointer(),2*NB,FO.pointer(),2*NB);
267
268 // Scatter Product into FDC
269 FDC = FO.template spinScatter<MatsT>();
270
271 }
272
273 }; // SingleSlater<T>::FDCommutator
