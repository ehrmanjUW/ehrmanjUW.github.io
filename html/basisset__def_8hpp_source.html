74 struct PrimShellHash
75 {
76 std::size_t operator()(libint2::Shell const& s) const noexcept
77 {
78 std::vector<size_t> hs{
79 std::hash<double>{}(s.alpha[0]),
80 std::hash<int>{}(s.contr[0].l),
81 std::hash<bool>{}(s.contr[0].pure),
82 std::hash<double>{}(s.O[0]),
83 std::hash<double>{}(s.O[1]),
84 std::hash<double>{}(s.O[2])
85 };
86 size_t h = 0;
87 for (size_t hi : hs)
88 h = hi ^ (h << 1);
89 return h;
90 }
91 };
