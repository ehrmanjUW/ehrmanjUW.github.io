36 template <typename IntsT>
37 template <typename IntsU>
38 std::shared_ptr<SquareMatrix<IntsU>>
39 SquareMatrix<IntsT>::convert(const std::shared_ptr<SquareMatrix<IntsT>> &mat) {
40
41 if (not mat) return nullptr;
42
43 const std::type_info &tID(typeid(*mat));
44
45 if (tID == typeid(SquareMatrix<IntsT>)) {
46 return std::make_shared<SquareMatrix<IntsU>>(*mat);
47
48 } else if (tID == typeid(PauliSpinorSquareMatrices<IntsT>)) {
49 return std::make_shared<PauliSpinorSquareMatrices<IntsU>>(
50 *std::dynamic_pointer_cast<PauliSpinorSquareMatrices<IntsT>>(mat));
51
52 } else {
53 std::stringstream errMsg;
54 errMsg << "SquareMatrix implementation \"" << tID.name()
55 << "\" not registered in convert." << std::endl;
56 CErr(errMsg.str(),std::cout);
57 }
58
59 return nullptr;
60
61 }
