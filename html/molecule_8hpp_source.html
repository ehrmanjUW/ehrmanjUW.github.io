126 // Function to save molecular coordinates, charge, multiplicity, and nuclei
127 void saveMolecule(){
128 std::vector<double> atom_coordinates;
129 std::vector<size_t> atomic_numbers;
130 int* mol_charge= &charge;
131 size_t* mol_multi= &multip;
132
133 for(size_t i=0;i<nAtoms;i++){
134 atomic_numbers.push_back(atoms[i].atomicNumber);
135 for( size_t j=0; j<3;j++){
136 atom_coordinates.push_back(atoms[i].coord[j]);
137 }
138 }
139 savfile.safeWriteData("/MOLECULE/COORDS",atom_coordinates.data(),{nAtoms,3});
140 savfile.safeWriteData("/MOLECULE/CHARGE",mol_charge,{1});
141 savfile.safeWriteData("/MOLECULE/MULTIP",mol_multi,{1});
142 savfile.safeWriteData("/MOLECULE/ATOMS",atomic_numbers.data(),{nAtoms,1});
143 }

164 inline void update() {
165
166 // Compute the total number of
167 nTotalE = std::accumulate(atoms.begin(),atoms.end(),-charge,
168 [&](int c, const Atom &a){ return a.atomicNumber + c; }
169 );
170
171 if(not ((nTotalE % 2) != 0 xor (multip % 2) != 0) or
172 multip > nTotalE + 1) {
173 std::stringstream ss;
174 ss << "Multiplicity = " << multip << " is not compatible with "
175 << "total electrons = " << nTotalE;
176 CErr(ss.str(),std::cout);
177 }
178
179 computeRIJ();
180 computeNNRep();
