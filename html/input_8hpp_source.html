169 static inline std::string& trim_left(std::string &s) {
170 s.erase(s.begin(), std::find_if(s.begin(), s.end(),
171 std::not1(std::ptr_fun<int, int>(std::isspace))));
172 return s;
173 }; // trim_left

181 static inline std::string& trim_right(std::string &s) {
182 s.erase(std::find_if(s.rbegin(), s.rend(),
183 std::not1(std::ptr_fun<int, int>(std::isspace))).base(), s.end());
184 return s;
185 }; // trim_right

193 static inline std::string &trim(std::string &s) {
194 return trim_left(trim_right(s));

205 static inline void split(std::vector<std::string>& tokens,
206 const std::string& str, const std::string& delimiters = " ") {
207
208 tokens.clear();
209 // Skip delimiters at beginning.
210 std::string::size_type lastPos = str.find_first_not_of(delimiters, 0);
211 // Find first "non-delimiter".
212 std::string::size_type pos = str.find_first_of(delimiters, lastPos);
213
214 while (std::string::npos != pos || std::string::npos != lastPos) {
215 // Found a token, add it to the vector.
216 tokens.push_back(str.substr(lastPos, pos - lastPos));
217 // Skip delimiters. Note the "not_of"
218 lastPos = str.find_first_not_of(delimiters, pos);
219 // Find next "non-delimiter"
220 pos = str.find_first_of(delimiters, lastPos);
221 }
222 }; // split
