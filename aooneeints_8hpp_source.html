45 template <typename IntsT>
46 void Integrals<IntsT>::computeAOOneE(CQMemManager &mem,
47 Molecule &mol, BasisSet &basis, EMPerturbation &emPert,
48 const std::vector<std::pair<OPERATOR,size_t>> &ops,
49 const AOIntsOptions &options) {
50
51 const std::array<std::string,3> dipoleList =
52 { "X","Y","Z" };
53 const std::array<std::string,6> quadrupoleList =
54 { "XX","XY","XZ","YY","YZ","ZZ" };
55 const std::array<std::string,10> octupoleList =
56 { "XXX","XXY","XXZ","XYY","XYZ","XZZ","YYY",
57 "YYZ","YZZ","ZZZ" };
58 const std::array<std::string,9> quadrupoleListAsymm =
59 { "XX","XY","XZ","YX","YY","YZ","ZX","ZY","ZZ" };
60
61 size_t NB = basis.nBasis;
62
63 for (const std::pair<OPERATOR,size_t> &op : ops)
64 switch (op.first) {
65 case OVERLAP:
66 overlap = std::make_shared<OneEInts<IntsT>>(mem, NB);
67 overlap->computeAOInts(basis, mol, emPert, OVERLAP, options);
68 if( savFile.exists() )
69 savFile.safeWriteData("INTS/OVERLAP", overlap->pointer(), {NB,NB});
70 break;
71
72 case KINETIC:
73 kinetic = std::make_shared<OneEInts<IntsT>>(mem, NB);
74 kinetic->computeAOInts(basis, mol, emPert, KINETIC, options);
75 if( savFile.exists() )
76 savFile.safeWriteData("INTS/KINETIC", kinetic->pointer(), {NB,NB});
77 break;
78
79 case NUCLEAR_POTENTIAL:
80 // Use Libint for point nuclei, in-house for Gaussian nuclei
81 if (options.OneEScalarRelativity)
82 potential = std::make_shared<OneERelInts<IntsT>>(
83 mem, NB, options.OneESpinOrbit);
84 else
85 potential = std::make_shared<OneEInts<IntsT>>(mem, NB);
86 potential->computeAOInts(basis, mol, emPert, NUCLEAR_POTENTIAL, options);
87 if( savFile.exists() ) {
88 std::string potentialTag = options.finiteWidthNuc ? "_FINITE_WIDTH" : "";
89 savFile.safeWriteData("INTS/POTENTIAL" + potentialTag,potential->pointer(),{NB,NB});
90 }
91 break;
92
93 case LEN_ELECTRIC_MULTIPOLE:
94 lenElectric = std::make_shared<MultipoleInts<IntsT>>(mem, NB, op.second, true);
95 lenElectric->computeAOInts(basis, mol, emPert, LEN_ELECTRIC_MULTIPOLE, options);
96 if( savFile.exists() ) {
97 // Length Gauge electric dipole
98 for(auto i = 0; i < 3; i++)
99 savFile.safeWriteData("INTS/ELEC_DIPOLE_LEN_" +
100 dipoleList[i], (*lenElectric)[dipoleList[i]].pointer(), {NB,NB} );
101
102 // Length Gauge electric quadrupole
103 if(op.second >= 2)
104 for(auto i = 0; i < 6; i++)
105 savFile.safeWriteData("INTS/ELEC_QUADRUPOLE_LEN_" +
106 quadrupoleList[i], (*lenElectric)[quadrupoleList[i]].pointer(), {NB,NB} );
107
108 // Length Gauge electric octupole
109 if(op.second >= 3)
110 for(auto i = 0; i < 10; i++)
111 savFile.safeWriteData("INTS/ELEC_OCTUPOLE_LEN_" +
112 octupoleList[i], (*lenElectric)[octupoleList[i]].pointer(), {NB,NB} );
113 }
114 break;
115
116 case VEL_ELECTRIC_MULTIPOLE:
117 velElectric = std::make_shared<MultipoleInts<IntsT>>(mem, NB, op.second, true);
118 velElectric->computeAOInts(basis, mol, emPert, VEL_ELECTRIC_MULTIPOLE, options);
119 if( savFile.exists() ) {
120 // Velocity Gauge electric dipole
121 for(auto i = 0; i < 3; i++)
122 savFile.safeWriteData("INTS/ELEC_DIPOLE_VEL_" +
123 dipoleList[i], (*velElectric)[dipoleList[i]].pointer(), {NB,NB} );
124
125 // Velocity Gauge electric quadrupole
126 if(op.second >= 2)
127 for(auto i = 0; i < 6; i++)
128 savFile.safeWriteData("INTS/ELEC_QUADRUPOLE_VEL_" +
129 quadrupoleList[i], (*velElectric)[quadrupoleList[i]].pointer(), {NB,NB} );
130
131 // Velocity Gauge electric octupole
132 if(op.second >= 3)
133 for(auto i = 0; i < 10; i++)
134 savFile.safeWriteData("INTS/ELEC_OCTUPOLE_VEL_" +
135 octupoleList[i], (*velElectric)[octupoleList[i]].pointer(), {NB,NB} );
136 }
137 break;
138
139 case MAGNETIC_MULTIPOLE:
140 magnetic = std::make_shared<MultipoleInts<IntsT>>(mem, NB, op.second, false);
141 magnetic->computeAOInts(basis, mol, emPert, MAGNETIC_MULTIPOLE, options);
142 if( savFile.exists() ) {
143 // Magnetic Dipole
144 for(auto i = 0; i < 3; i++)
145 savFile.safeWriteData("INTS/MAG_DIPOLE_" +
146 dipoleList[i], (*magnetic)[dipoleList[i]].pointer(), {NB,NB} );
147
148 // Magnetic Quadrupole
149 if(op.second >= 2)
150 for(auto i = 0; i < 6; i++)
151 savFile.safeWriteData("INTS/MAG_QUADRUPOLE_" +
152 quadrupoleList[i], (*magnetic)[i+3].pointer(), {NB,NB} );
153 }
154 break;
155
156 case ELECTRON_REPULSION:
157 CErr("Electron repulsion integrals are not implemented in computeAOOneE,"
158 " they are implemented in TwoEInts",std::cout);
159 break;
160 }
161
162 }; // AOIntegrals<IntsT>::computeAOOneE
