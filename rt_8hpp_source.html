131 RunChronusQ(TEST_ROOT #in ".inp","STDOUT", \

140 auto energyDim1 = resFile.getDims("/RT/ENERGY");\
141 auto energyDim2 = refFile.getDims("/RT/ENERGY");\
142 ASSERT_EQ( energyDim1.size(), 1 );\
143 ASSERT_EQ( energyDim2.size(), 1 );\
144 ASSERT_EQ( energyDim1[0], energyDim2[0] );\
145 \

149 ASSERT_EQ( dipoleDim2.size(), 2 );\

152 ASSERT_EQ( dipoleDim2[1], 3 );\

159 EXPECT_NEAR(xDummy[i], yDummy[i], tol);\
