50 void QuantumBase::printMultipoles(std::ostream &out) {
51
52 out << std::left << std::setprecision(10);
53 double xSmall = 1e-10;
54 for(auto i = 0; i < 3; i++) {
55 if(std::abs(this->elecDipole[i]) < xSmall)
56 this->elecDipole[i] = std::abs(this->elecDipole[i]);
57
58 for(auto j = 0; j < 3; j++) {
59 if(std::abs(this->elecQuadrupole[i][j]) < xSmall)
60 this->elecQuadrupole[i][j] = std::abs(this->elecQuadrupole[i][j]);
61
62 for(auto k = 0; k < 3; k++)
63 if(std::abs(this->elecOctupole[i][j][k]) < xSmall)
64 this->elecOctupole[i][j][k] =
65 std::abs(this->elecOctupole[i][j][k]);
66
67 }
