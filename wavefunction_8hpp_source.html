87 WaveFunction(MPI_Comm c, CQMemManager &mem, Molecule &mol, size_t nBasis,
88 Integrals<IntsT> &aoi, size_t _nC, bool iCS) :
89 QuantumBase(c, mem,_nC,iCS),
90 WaveFunctionBase(c, mem,_nC,iCS),
91 Quantum<MatsT>(c, mem,_nC,iCS,nBasis),
92 molecule_(mol), eps1(nullptr), eps2(nullptr), aoints(aoi) {
93
94 // Compute meta data
95
96 this->nO = molecule_.nTotalE;
97 this->nV = 2*nBasis - nO;
98
99 if( this->iCS ) {
100 this->nOA = this->nO / 2; this->nOB = this->nO / 2;
101 this->nVA = this->nV / 2; this->nVB = this->nV / 2;
102 } else {
103 size_t nSingleE = molecule_.multip - 1;
104 this->nOB = (this->nO - nSingleE) / 2;
105 this->nOA = this->nOB + nSingleE;
106 this->nVA = nBasis - this->nOA;
107 this->nVB = nBasis - this->nOB;
108 }
109
110 // Allocate internal memory
111 if(nBasis != 0) alloc();
112
113 };
