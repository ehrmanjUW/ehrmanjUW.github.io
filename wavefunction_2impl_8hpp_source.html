49 template <typename MatsT, typename IntsT>
50 template <typename MatsU>
51 WaveFunction<MatsT,IntsT>::WaveFunction(const WaveFunction<MatsU,IntsT> &other,int dummy) :
52 aoints(other.aoints),
53 QuantumBase(dynamic_cast<const QuantumBase &>(other)),
54 WaveFunctionBase(dynamic_cast<const WaveFunctionBase &>(other)),
55 Quantum<MatsT>(dynamic_cast<const Quantum<MatsU>&>(other)),
56 molecule_(other.molecule_) {
57
58 mo.reserve(2);
59 for (const SquareMatrix<MatsU> &mat : other.mo)
60 mo.emplace_back(mat);
61
62#ifdef _WaveFunctionDebug
63 std::cout << "WaveFunction<T>::WaveFunction(const WaveFunction<U>&) "
64 << "(this = " << this << ", other = " << &other << ")"
65 << std::endl;
66#endif
67 WaveFunction_COLLECTIVE_OP(COPY_OTHER_MEMBER,COPY_OTHER_MEMBER_OP);
68
69 }; // WaveFunction<T>::WaveFunction(const WaveFunction<U> &)

119 template <typename MatsT, typename IntsT>
120 void WaveFunction<MatsT,IntsT>::alloc() {
121
122#ifdef _WaveFunctionDebug
123 std::cout << "WaveFunction::alloc (this = " << this << ")" << std::endl;
124#endif
125
126 size_t NB = this->nC * this->nAlphaOrbital();
127
128 mo.reserve(2);
129 mo.emplace_back(memManager, NB);
130 eps1 = memManager.malloc<double>(NB);
131
132 if( this->nC == 1 and (not this->iCS) ) {
133 mo.emplace_back(memManager, NB);
134 eps2 = this->memManager.template malloc<double>(NB);
135 }
136
137 }; // WaveFunction<T>::alloc
