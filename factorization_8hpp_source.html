187 template <typename _F>
188 inline int QZ(char JOBVSL, char JOBVSR, int N, _F *A, int LDA, _F *B, int LDB,
189 dcomplex *W, _F *VSL, int LDVSL, _F *VSR, int LDVSR, CQMemManager &mem) {
190
191
192 dcomplex *ALPHA = mem.template malloc<dcomplex>(N);
193 _F *BETA = mem.template malloc<_F>(N);
194
195 int INFO = QZ(JOBVSL,JOBVSR,N,A,LDA,B,LDB,ALPHA,BETA,VSL,LDVSL,VSR,LDVSR,
196 mem);
197
198 for(auto k = 0; k < N; k++) {
199
200 if( std::abs(BETA[k]) < 1e-12 )
201 std::cout << "WARNING: SMALL BETA INCURRED IN QZ" << std::endl;
202
203 W[k] = ALPHA[k] / BETA[k];
204
205 }
206
207 mem.free(ALPHA,BETA);
208
209 return INFO;
210
211 }
