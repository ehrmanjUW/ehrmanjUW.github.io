62 inline void fixBlockNumber() {
63 if(N_ % BlockSize_) {
64 #ifdef MEM_PRINT
65 std::cerr << "Memory Block not Divisible by BlockSize ("
66 << BlockSize_ << " Bytes). Increasing allocation by "
67 << BlockSize_ - (N_ % BlockSize_) << " Bytes" << std::endl;

69 #endif
70 N_ += BlockSize_ - (N_ % BlockSize_);

95 CQMemManager(size_t N = 0, size_t BlockSize = 2048) :
96 mem_backend(), N_(N), BlockSize_(BlockSize), isAllocated_(false),
97 NAlloc_(0) {
98 if( N_ and BlockSize_ ) allocMem();
99 };

104 void allocMem() {
105 assert(not isAllocated_);
106
107 fixBlockNumber(); // fix buffer length
108 V_.resize(N_); // allocate the memory
109
110 // segregate (uses functionality from boost::simple_segregated_storage
111 this->add_ordered_block(&V_.front(),V_.size(),BlockSize_);
112
113 isAllocated_ = true; // ensure no additional allocs can be made
114
115 #ifdef MEM_PRINT
116 std::cerr << "Creating Memory Partition of " << N_
117 << " bytes starting at " << (int*) &V_.front()
118 << std::endl;
119 #endif
120 };

134 template <typename T>
135 T* malloc(size_t n) {
136 // Determine the number of blocks to allocate
137 size_t nBlocks = ( (n-1) * sizeof(T) ) / BlockSize_ + 1;
138
139 // Check to see if requested memory would overflow allocated
140 // memory. Throw a bad alloc if so
141 if( (NAlloc_ + nBlocks) * BlockSize_ > N_ ) {
142 std::bad_alloc excp;
143 throw excp;
144 }
145
146 // Update the number of allocated blocks
147 NAlloc_ += nBlocks;
148
149 #ifdef MEM_PRINT
150 std::cerr << "Allocating " << n << " words of " << typeid(T).name()
151 << " data (" << nBlocks << " blocks): ";
152 #endif
153
154 // Get a pointer from boost::simple_segregated_storage
155 void * ptr = mem_backend::malloc_n(nBlocks,BlockSize_);
156
157 // Throw an error if boost returned a NULL pointer (many
158 // possible causes)
159 if(ptr == NULL) {
160 std::bad_alloc excp;
161 throw excp;
162 }
163
164 #ifdef MEM_PRINT
165 std::cerr << " PTR = " << ptr << std::endl;
166 #endif
167
168 assert( AllocatedBlocks_.find(static_cast<void*>(ptr)) == AllocatedBlocks_.end() );
169
170 // Keep a record of the block
171 AllocatedBlocks_[ptr] = { n * sizeof(T), nBlocks };
172
173 return static_cast<T*>(ptr); // Return the pointer
174 }; // CQMemManager::malloc

186 template <typename T>
187 void free( T* &ptr ) {
188
189 // Attempt to find the pointer in the list of
190 // allocated blocks
191 auto it = AllocatedBlocks_.find(static_cast<void*>(ptr));
192
193 // Kill the job if the pointer is not in the list of allocated
194 // blocks
195 assert( it != AllocatedBlocks_.end() );
196
197 #ifdef MEM_PRINT
198 std::cerr << "Freeing " << it->second.second
199 << " blocks of data starting from "
200 << static_cast<void*>(ptr) << std::endl;
201 #endif
202
203 NAlloc_ -= it->second.second; // deduct block size from allocated memory
204
205 // deallocate the memory in an ordered fashion
206 mem_backend::ordered_free_n(ptr,it->second.second,BlockSize_);
207
208 // Remove pointer from allocated list
209 AllocatedBlocks_.erase(it);
210
211 ptr = NULL; // NULL out the pointer
212 }; // CQMemManager::free

243 template <typename T>
244 size_t getSize(T* ptr) {
245 // Attempt to find the pointer in the list of
246 // allocated blocks
247 auto it = AllocatedBlocks_.find(static_cast<void*>(ptr));
248
249 // Kill the job if the pointer is not in the list of allocated
250 // blocks
251 assert( it != AllocatedBlocks_.end() );
252
253 return std::floor(it->second.first / sizeof(T));
254 }; // CQMemManager::getSize
