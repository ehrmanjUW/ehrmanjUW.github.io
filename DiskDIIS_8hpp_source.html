153 // Write current solution vector to disk
154 template <typename T>
155 void DiskDIIS<T>::write_vector(T * vector){
156
157 // Name the entry in according to the current DiskDIIS iteration.
158 // If we already have maxdiis_ vectors, then replace one.
159
160 char * oldvector = memManager_.template malloc<char>(1000);
161
162 if ( diis_iter_ <= maxdiis_ ){
163 sprintf(oldvector,"/DIIS/VECTOR%i",diis_iter_);
164 }
165 else{
166 sprintf(oldvector,"/DIIS/VECTOR%i",replace_diis_iter_);
167 }
168
169 // Write the current solution vector.
170 savFile_.safeWriteData(oldvector, vector, {dimdiis_});
171
172 memManager_.free(oldvector);
173 }

192 void DiskDIIS<T>::write_error_vector(T * vector){
193
194 // Name the entry in according to the current DiskDIIS iteration.
195 // If we already have maxdiis_ vectors, then replace one.
196
197 char * evector = memManager_.template malloc<char>(1000);
198 if ( diis_iter_ <= maxdiis_ ){
199 sprintf(evector,"/DIIS/ERROR%i",diis_iter_);
200 }
201 else{
202 sprintf(evector,"/DIIS/ERROR%i",replace_diis_iter_);
203 }
204
205 if ( diis_iter_ == 0 ) {
206 // On the first iteration, write an entry that will hold the error matrix
207 T * temp = (T*)malloc(maxdiis_*maxdiis_*sizeof(T));
208 memset((void*)temp,'\0',maxdiis_*maxdiis_*sizeof(T));
209 savFile_.safeWriteData("/DIIS/ERROR_MATRIX", temp, {maxdiis_*maxdiis_});
210 free(temp);
211 }
212
213 // Write current error vector.
214 savFile_.safeWriteData(evector, vector, {dimdiis_});
215
216 memManager_.free(evector);

280 // Evaluate extrapolation coefficients for DiskDIIS.
281 template <typename T>
282 void DiskDIIS<T>::DIISCoefficients(int nvec){
283
284 // Allocate memory for small matrices/vectors.
285 int * ipiv = memManager_.template malloc<int>(nvec+1);
286
287 T * temp = memManager_.template malloc<T>(maxdiis_*maxdiis_);
288 T * A = memManager_.template malloc<T>((nvec+1)*(nvec+1));
289 T * B = memManager_.template malloc<T>(nvec+1);
290
291 memset((void*)A,'\0',(nvec+1)*(nvec+1)*sizeof(T));
292 memset((void*)B,'\0',(nvec+1)*sizeof(T));
293
294 B[nvec] = -1.0;
295
296 char * evector = memManager_.template malloc<char>(1000);
297
298 // Read in the previous error matrix, so we don't have
299 // to build the entire thing each iteration.
300
301 savFile_.readData("/DIIS/ERROR_MATRIX",temp);
302
303 // Reshape the error matrix, in case its dimension is less than maxdiis_.
304 for (int i = 0; i < nvec; i++){
305 for (int j = 0; j < nvec; j++){
306 A[i*(nvec+1)+j] = temp[i*maxdiis_+j];
307 }
308 }
309
310 if (nvec <= 3) {
311 // At early iterations, just build the whole matrix.
312 for (int i = 0; i < nvec; i++) {
313 sprintf(evector,"/DIIS/ERROR%i",i+1);
314 savFile_.readData(evector,tmp1_);
315 for (int j = i+1; j < nvec; j++){
316 sprintf(evector,"/DIIS/ERROR%i",j+1);
317 savFile_.readData(evector,tmp2_);
318 T sum = InnerProd<T>(dimdiis_,tmp1_,1,tmp2_,1);
319 A[i*(nvec+1)+j] = sum;
320 A[j*(nvec+1)+i] = sum;
321 }
322 T sum = InnerProd<T>(dimdiis_,tmp1_,1,tmp1_,1);
323 A[i*(nvec+1)+i] = sum;
324 }
325 }else {
326 // At later iterations, don't build the whote matrix.
327 // Just replace one row/column.
328
329 // Which row/column will be replaced?
330 int i = nvec < maxdiis_ ? nvec - 1 : replace_diis_iter_ - 1;
331
332 sprintf(evector,"/DIIS/ERROR%i",i+1);
333 savFile_.readData(evector,tmp1_);
334 for (int j = 0; j < nvec; j++){
335 sprintf(evector,"/DIIS/ERROR%i",j+1);
336 savFile_.readData(evector,tmp2_);
337 T sum = InnerProd<T>(dimdiis_,tmp1_,1,tmp2_,1);
338 A[i*(nvec+1)+j] = sum;
339 A[j*(nvec+1)+i] = sum;
340 }
341 }
342
343 int j = nvec;
344 for (int i = 0; i < (nvec+1); i++){
345 A[j*(nvec+1)+i] = -1.0;
346 A[i*(nvec+1)+j] = -1.0;
347 }
348 A[(nvec+1)*(nvec+1)-1] = 0.0;
349
350 // Write error matrix for next iteration.
351 for (int i = 0; i < nvec; i++){
352 for (int j = 0; j < nvec; j++){
353 temp[i*maxdiis_+j] = A[i*(nvec+1)+j];
354 }
355 }
356 savFile_.safeWriteData("/DIIS/ERROR_MATRIX", temp, {maxdiis_*maxdiis_});
357
358 memManager_.free(temp);
359 memManager_.free(evector);
360
361 // Solve the set of linear equations for the extrapolation coefficients
362 int nrhs = 1;
363 int lda = nvec+1;
364 int ldb = nvec+1;
365 LinSolve<T>(nvec+1,nrhs,A,lda,B,ldb,ipiv);
366 std::copy_n(B,nvec,diisvec_);
367
368 memManager_.free(A);
369 memManager_.free(B);
370 memManager_.free(ipiv);
371 }
