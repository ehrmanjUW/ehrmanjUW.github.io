235 inline virtual void run() {
236
237 // Synchronize MPI processes
238 MPI_Barrier(comm_);
239
240 if( alreadyRan_ ) CErr("RESPONSE module only designed to be run once!");
241
242 genSettings.matIsHer = genSettings.doTDA; // TDA always implies hermetian matrix
243 configOptions();
244
245 bool rootHasFullMat = (MPIRank(comm_) == 0) and fullMatrix_;
246#ifdef CQ_ENABLE_MPI
247 MPIBCast(&rootHasFullMat,1,0,comm_);
248#endif
249
250 writeMeta();
251
252 if( (genSettings.doFull or genSettings.formFullMat) and
253 not rootHasFullMat ) formFullMatrix();
254
255
256 allocResults();
257
258 bool isRoot = MPIRank(comm_) == 0;
259
260 if( isRoot and genSettings.printLevel >= 0 )
261 std::cout << " ** STARTING RESPONSE CALCULATION\n\n";
262
263 auto topResp = tick();
264
265 if( genSettings.jobType == RESIDUE ) runResidue();
266 else runFDR();
267
268 double botResp = tock(topResp);
269
270 if( isRoot and genSettings.printLevel >= 0 )
271 std::cout << "\n\n ** RESPONSE CALCULATION FINISHED ("
272 << std::fixed << botResp << " s)\n";
273
274
275 // Synchronize MPI processes
276 MPI_Barrier(comm_);
277 alreadyRan_ = true;
278
279 };

314 virtual void postLinearSolve(){
315
316
317 ROOT_ONLY(comm_);
318
319 bool noDamp = (fdrSettings.dampFactor == 0.) and
320 not fdrSettings.forceDamp;
321
322 size_t N = nSingleDim_;
323 size_t nOmega = fdrSettings.bFreq.size();
324 size_t nRHS = fdrSettings.nRHS;
325
326 // Write Solutions to disk
327 if( savFile.exists() ) {
328
329 savFile.safeWriteData("/RESP/FDR/OMEGA",&fdrSettings.bFreq[0],
330 {nOmega});
331
332
333 savFile.safeWriteData("/RESP/FDR/DAMP",&fdrSettings.dampFactor,
334 {1});
335
336
337 if( noDamp ) {
338
339 savFile.safeWriteData("/RESP/FDR/RHS",fdrResults.RHS, {nRHS,N});
340
341 savFile.safeWriteData("/RESP/FDR/SOLUTION",fdrResults.SOL,
342 {nRHS*nOmega,N});
343
344 } else {
345
346 savFile.safeWriteData("/RESP/FDR/RHS",dfdrResults.RHS, {nRHS,N});
347
348 savFile.safeWriteData("/RESP/FDR/SOLUTION",dfdrResults.SOL,
349 {nRHS*nOmega,N});
350 }
351
352 };

576 template <typename T>
577 template <typename U>
578 void ResponseTBase<T>::iterLinearTrans(size_t nVec, U* V, U* AV) {
579
580 std::vector<RESPONSE_CONTRACTION<U>> cList(1);
581 cList.back().nVec = nVec;
582 cList.back().N = nSingleDim_;
583 cList.back().X = V;
584 cList.back().AX = AV;
585
586#ifdef CQ_ENABLE_MPI
587 CB_INT MLoc, NLoc;
588 if( genSettings.isDist() ) {
589 std::tie(MLoc,NLoc) = fullMatGrid_->getLocalDims(nSingleDim_,nVec);
590 if( MLoc and NLoc ) {
591
592 cList.back().X = this->memManager_.template malloc<U>(MLoc*NLoc);
593 cList.back().AX = this->memManager_.template malloc<U>(MLoc*NLoc);
594
595 } else {
596
597 cList.back().X = nullptr;
598 cList.back().AX = nullptr;
599
600 }
601
602 cList.back().DescX = fullMatGrid_->descInit(nSingleDim_,nVec,0,0,MLoc);
603 cList.back().DescAX = cList.back().DescX;
604
605 fullMatGrid_->Scatter(nSingleDim_,nVec,V,nSingleDim_,cList.back().X,
606 MLoc,0,0);
607 }
608#endif
609
610 this->formLinearTrans( cList );
611
612#ifdef CQ_ENABLE_MPI
613 if( genSettings.isDist() ) {
614
615 fullMatGrid_->Gather(nSingleDim_,nVec,AV,nSingleDim_,cList.back().AX,
616 MLoc,0,0);
617
618 if( cList.back().X ) this->memManager_.free(cList.back().X );
619 if( cList.back().AX ) this->memManager_.free(cList.back().AX);
620
621 }
622#endif
623
624 }
